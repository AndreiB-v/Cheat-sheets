# Копирование репозитория по сслыке
git clone <repository_url> [<local path name>]

# Инициализация Git в папке (создание .git)
git init

# Привязка глобального репозитория
git remote add origin <repository_url>

# Добавление файлов в следующий коммит
git add <filename>
git add .

git commit -m 'commit text'

# Опция -u (или --set-upstream): в будущем, при выполнении git push, Git будет знать, с какой удаленной веткой синхронизировать вашу локальную ветку. При смене ветки нужно будет повторить операцию.
git push -u origin <your branch>

# Принудительное переименование текущей ветки
git branch -M main
