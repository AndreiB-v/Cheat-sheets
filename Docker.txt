============================ Работа с образами ============================

# Загрузка образа
docker pull <image_name>

# Список доступных для скачивания образов
docker search <image_name>


========================== Работа с контейнерами ==========================

# Создание и запуск контейнера из образа
# Пример: ~$ docker run -it --name first ubuntu:focal
docker run <image_name>
docker container run <image_name>
# Если образ не скачан, то скачется из Docker Hub (по умолчанию)
# -it — позволяет взаимодействовать с контейнером через текущую командную оболочку
# --name — поможет вам задать имя получившемуся контейнеру
# sleep — запуск работы контейнера в течении N секунд
# -d (detached mode) — фоновый запуск контейнера

# Повторный запуск уже существующего контейнера
docker start <CONTAINER_ID>

# Остановка контейнера
docker stop <container_name>

# Удаление контейнера
docker rm <container_name>


=============================== Информация ===============================

# Список контейнеров (Docker process status)
docker container ls
docker ps
# -a - выводит информацию о ранее запущенных контейнерах
# CONTAINER ID: уникальный идентификатор контейнера
# IMAGE: образ, на основе которого создан контейнер
# COMMAND: команда, запущенная внутри контейнера
# CREATED: время, когда контейнер был создан
# STATUS: статус контейнера (например, Up для работающих контейнеров или Exited для остановленных)
# PORTS: порты, проброшенные из контейнера в хостовую систему (если есть)
# NAMES: имя контейнера (по умолчанию Docker генерирует случайное имя, но вы можете задать имя контейнера при создании)

# Список Docker-образов
docker images
docker image ls
# REPOSITORY: имя репозитория образа
# TAG: тег образа, который указывает на конкретную версию или вариант образа (например, версия ОС или приложения)
# IMAGE ID: уникальный идентификатор образа
# CREATED: дата создания образа
# SIZE: размер образа в байтах

# Просмотр информации об образе
docker image inspect <image_name:tag / image_id>
