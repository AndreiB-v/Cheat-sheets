============== Инициализация и подключение к репозиторию ==============

# Инициализация Git в папке (создание .git)
git init

# Копирование репозитория по сслыке
git clone <repository_url> <local_path_name (не обязательно)>

# Привязка глобального репозитория
git remote add origin <repository_url>



========================== Работа с коммитами ==========================

# Добавление файлов в индекс
git add <filename>
git add .

# -m - комментарий
git commit -m 'commit text'

git push -u origin <branch_name>
# Опция -u (или --set-upstream): запомнить ветку. В дальнейшем просто git push. При смене ветке повторить операцию
# origin --delete <branch_name> - удаление ветки на удаленном репозитории

git pull -u <repository_url (origin по умолчанию)> <branch (по умолчанию активная ветка)>
# --set-upstream - запомнить выбор

# Принудительное переименование текущей ветки
git branch -M main

# Отмена изменений
git reset <commit>
# Аргумент: HEAD~1 — один коммит назад от HEAD (текущий коммит), HEAD~2 - два коммита и т.д
# --mixed (по умолчанию) - Этот вариант сбрасывает последний коммит к указанному коммиту и снимает изменения из индекса
# --soft - Этот вариант сбрасывает последний коммит к указанному коммиту, но сохраняет изменения в вашей рабочей директории и индексе
# --hard - Этот вариант сбрасывает последний коммит к указанному коммиту, удаляет изменения из индекса и рабочей директории



=============================== Информация ===============================

git status
# Текущая ветка
# Измененные файлы
# Подготовленные к коммиту файлы

git log
# История коммитов

git remote
# Просмотр списка удаленных репозиториев
# -v - просмотр адресов репозиториев

git fetch <branch>
# Позволяет вам смотреть, какие изменения произошли в удаленном репозитории без изменения вашей текущей рабочей директории
# branch — параметр, который определяет удаленную ветку, из которой вы хотите получить изменения. Если он не указан, то по умолчанию используется текущая ветка, с которой вы работаете локально.


============================= Работа с ветками =============================

git branch <branch_name>
# По умолчанию - создание новой ветки
# Без параметра - список всех локальных веток в вашем репозитории. Текущая помечена звездочкой
# -D - удаление ветки

git checkout <branch_name>
# Переключение на ветку
# -b - с созданием новой ветки с этим названием

git merge <branch_name>
# Слияние веток
# branch_name — название ветки, которую вы хотите интегрировать в текущую
